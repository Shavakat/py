git clone (ссылка на репозиторий)   #создание локального репозитория
touch 1.py                          #создание python файла
git add 1.py                        #индексирование файла
git commit -m '(текст любой)'       #сохранение состояния файла
git push                            #загрузка изменений в центральный репозиторий
git pull                            #загрузка изменений из центрального репозитория
python3 1.py или python 1.py        #запуск python файла
